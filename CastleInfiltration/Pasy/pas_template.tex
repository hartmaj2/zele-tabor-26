% --- Makro pro podmíněnou položku zvláštního znamení ----------------
% Vykreslí bullet s textem, jen pokud je text neprázdný.
\newcommand{\znameniItem}[1]{%
  \edef\tmp{#1}%
  \ifx\tmp\empty\else
    \item[\textcolor{cernarudna}{$\bullet$}] #1%
  \fi
}

% --- Makro pro pole ve formátu "popisek nad hodnotou" ----------------
\newcommand{\polePas}[2]{%
  {\fontPolePopisek\color{popisek}\textsc{#1}}\par
  \vspace{\dataMezeraNadHodnotou}%
  {\fontPoleHodnota\bfseries\color{cernarudna} #2}\par
  \vspace{\dataMezeraMeziPoli}%
}
