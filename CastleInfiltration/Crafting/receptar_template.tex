\newcommand{\RecipeBookTitle}{\detokenize{__BOOK_TITLE__}}

\newcommand{\recipeCard}[3]{%
  \begin{tikzpicture}
    \node[draw=gridline, fill=paper,
          line width=\ParamCardBorderPt pt,
          rounded corners=\ParamCardCorner,
          inner sep=\ParamCardPaddingMM mm,
          text width=\ParamCardContentWidth] (card) {%
      \begingroup\ParamNameSize\bfseries\color{ink}\ParamNameAlign #1\par\endgroup
      \vspace{1.5mm}
      \begin{center}#3\end{center}
      \vspace{1mm}
      {\footnotesize\color{muted} #2\par}%
    };
  \end{tikzpicture}%
}
